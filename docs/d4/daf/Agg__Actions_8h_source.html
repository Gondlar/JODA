110 struct aggExpAction<aggSingleExp> {
111  template <typename Input>
112  static void apply(const Input &in, aggState &state) {
113  assert(state.aggfun != NOAGG);
114  std::unique_ptr<joda::query::IAggregator> agg = nullptr;
115  try {
116  switch (state.aggfun) {
117  case NOAGG:
118  assert(false);
119  break;
120  case AVG:
121  agg = std::make_unique<joda::query::AverageAggregator>(
122  state.toPointer, std::move(state.valprov));
123  break;
124  case COUNT:
125  agg = std::make_unique<joda::query::CountAggregator>(
126  state.toPointer, std::move(state.valprov));
127  break;
128  case SUM:
129  agg = std::make_unique<joda::query::SumAggregator>(
130  state.toPointer, std::move(state.valprov));
131  break;
132  case ATTSTAT:
133  agg = std::make_unique<joda::query::AttributeStatAggregator>(
134  state.toPointer, std::move(state.valprov));
135  break;
136  case DISTINCT:
137  agg = std::make_unique<joda::query::DistinctAggregator>(
138  state.toPointer, std::move(state.valprov));
139  break;
140  case MIN:
141  agg = std::make_unique<joda::query::MinAggregator>(
142  state.toPointer, std::move(state.valprov));
143  break;
144  case MAX:
145  agg = std::make_unique<joda::query::MaxAggregator>(
146  state.toPointer, std::move(state.valprov));
147  break;
148  case COLLECT:
149  agg = std::make_unique<joda::query::CollectAggregator>(
150  state.toPointer, std::move(state.valprov));
151  break;
152  }
153  } catch (joda::query::WrongParameterException &e) {
154  throw tao::pegtl::parse_error(e.what(), in);
155  }
156  DCHECK(agg != nullptr);
157  if (state.groupedByValue == nullptr) { // Normal aggregation
158  state.aggs.push_back(std::move(agg));
159  } else { // Group by aggregation
160  auto groupAgg = std::make_unique<joda::query::GroupAggregator>(
161  state.toPointer, std::move(state.groupedByValue), std::move(agg));
162  if (!state.groupAs.empty()) {
163  groupAgg->setGroupAs(state.groupAs);
164  state.groupAs.clear();
165  }
166  state.aggs.push_back(std::move(groupAgg));
167  }
168  DCHECK(agg == nullptr);
169  DCHECK(state.groupedByValue == nullptr);
170  state.valprov.clear();
171  state.toPointer.clear();
172  state.aggfun = NOAGG;
173  assert(state.valprov.empty());
174  assert(state.toPointer.empty());
175  assert(state.aggfun == NOAGG);
176  }
177 };
