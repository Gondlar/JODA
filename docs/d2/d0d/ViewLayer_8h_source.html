123  auto members = GetMembers(curr);
124  auto baseAttPath = curr + "/";
125  auto baseAttSize = baseAttPath.size();
126  if (RAPIDJSON_UNLIKELY(!handler.StartObject())) return false;

131 
132  if (vos_enabled) {
133  obj.reserve(members.size());
134  }
135  for (const auto &member : members) {
136  if (RAPIDJSON_UNLIKELY(
137  !handler.Key(member.name.c_str(), member.name.size(), true)))
138  return false;
139  if (member.val != nullptr) {
140  if (vos_enabled)
141  obj.addMember(struc->getOrAdd(member.name), member.val);
142  if (RAPIDJSON_UNLIKELY(!member.val->Accept(handler))) return false;
143  } else {
144  cursor->followAtt(member.name);
145  baseAttPath.erase(baseAttSize);

147  if (!Accept(handler, baseAttPath)) {
148  return false;

152  obj.addMember(struc->getOrAdd(member.name), &objects[baseAttPath]);
153  }
